% =========================================================
% 标题编号与标题格式
%
% \section      → 一、一级标题

\renewcommand{\thesection}{\chinese{section}}

  {\firsttitlefont\mdseries\fontsize{16pt}{28pt}\selectfont}
  {\thesection、}

  {}

\titleformat{\subsection}[hang]
  {\secondtitlefont\mdseries\fontsize{16pt}{28pt}\selectfont}

  {\thirdtitlefont\bfseries\fontsize{16pt}{28pt}\selectfont}

% =========================================================
% 页眉和页脚
%
% 默认状态：
%   页眉为空；
%   自动显示页码；
%   奇数页页码位于正文版心右侧；
%   偶数页页码位于正文版心左侧；
%   页码格式为“— 1 —”“— 2 —”“— 3 —”；
%   页码字体为宋体，字号为四号（14 pt）。
%
% 自定义命令：
%   \SetHeader{页眉内容}                 % 添加或修改页眉
%   \ClearHeader                         % 清空页眉
%   \SetPageNumberFormat{第\thepage 页}  % 修改页码格式
%   \HidePageNumber                      % 隐藏页码
%   \ShowPageNumber                      % 恢复页码
%
% 页眉基线距纸张上边缘 1.5 cm；
% 页码基线距纸张下边缘 1.75 cm。
% =========================================================

% 关闭 ctexart/LaTeX 自带的页眉页脚，由本模板统一绘制
\pagestyle{empty}

% 页眉默认留空
\newcommand{\HeaderText}{}
\newcommand{\SetHeader}[1]{\renewcommand{\HeaderText}{#1}}
\newcommand{\ClearHeader}{\renewcommand{\HeaderText}{}}

% 页码默认开启，格式为“— 1 —”
\newif\ifshowpagenumber
\showpagenumbertrue
\newcommand{\PageNumberText}{\mbox{—\hspace{0.5em}\thepage\hspace{0.5em}—}}
\newcommand{\SetPageNumberFormat}[1]{\renewcommand{\PageNumberText}{#1}}
\newcommand{\HidePageNumber}{\showpagenumberfalse}
\newcommand{\ShowPageNumber}{\showpagenumbertrue}

\AddToShipoutPictureFG{%
  % 自定义页眉：默认为空
  \AtPageUpperLeft{%
    \put(
      \LenToUnit{0.5\paperwidth},
      -\LenToUnit{1.5cm}
    ){%
      \makebox(0,0)[c]{%
        \fontsize{10.5pt}{12pt}\selectfont\HeaderText
      }%
    }%
  }%
  % 页码：奇数页右对齐，偶数页左对齐
  \AtPageLowerLeft{%
    \ifshowpagenumber
      \ifodd\value{page}
        \put(
          \LenToUnit{2.8cm},
          \LenToUnit{1.75cm}
        ){%
          \makebox(\LenToUnit{\textwidth},0)[r]{%
            \pagenumberfont\fontsize{14pt}{14pt}\selectfont
            \PageNumberText
          }%
        }%
      \else
        \put(
          \LenToUnit{2.8cm},
          \LenToUnit{1.75cm}
        ){%
          \makebox(\LenToUnit{\textwidth},0)[l]{%
            \pagenumberfont\fontsize{14pt}{14pt}\selectfont
            \PageNumberText
          }%
        }%
      \fi
    \fi
  }%
}

% =========================================================
% 可选公文式目录
%
% 使用方法：
%   在 \begin{document} 后、正文主标题前加入：
%       \MakeOfficialTOC
%
% 若不加入该命令，则不会生成目录。
%
% 目录页规则：
%   1. 目录位于本文件第一页；
%   2. 目录页不显示页码，也不计入正文页码；
%   3. 正文首页重新从“— 1 —”开始编号；
%   4. 目录列出一级、二级和三级标题；
%   5. 新增或修改标题后，需要编译两次，或直接使用 latexmk。
% =========================================================

% 目录标题：方正小标宋简体、二号、32 pt 行距、居中
\newcommand{\OfficialTOCTitle}{%
  \begin{center}
    {\maintitlefont\mdseries\fontsize{22pt}{32pt}\selectfont
      目\hspace{1em}录}
  \end{center}
  \vspace{16pt}
}

% 目录最多显示三级标题
\setcounter{tocdepth}{3}

% 一级目录：黑体、三号，顶格
\titlecontents{section}
  [0pt]
  {\firsttitlefont\mdseries\fontsize{16pt}{28pt}\selectfont}
  {\contentslabel[\thecontentslabel、]{2.5em}}
  {}
  {\bodyfont\mdseries\titlerule*[0.5pc]{.}\contentspage}

% 二级目录：楷体_GB2312、三号，缩进两个汉字
\titlecontents{subsection}
  [2em]
  {\secondtitlefont\mdseries\fontsize{16pt}{28pt}\selectfont}
  {\contentslabel[（\thecontentslabel）]{3.5em}}
  {}
  {\bodyfont\mdseries\titlerule*[0.5pc]{.}\contentspage}

% 三级目录：仿宋_GB2312、三号，缩进四个汉字
\titlecontents{subsubsection}
  [4em]
  {\bodyfont\mdseries\fontsize{16pt}{28pt}\selectfont}
  {\contentslabel[\thecontentslabel.]{2.5em}}
  {}
  {\bodyfont\mdseries\titlerule*[0.5pc]{.}\contentspage}

\makeatletter
\newcommand{\MakeOfficialTOC}{%
  \clearpage
  \begingroup
    % 目录页临时取消页眉和页码；正文设置不受影响
    \renewcommand{\HeaderText}{}%
    \HidePageNumber
    \pagenumbering{gobble}%
    \thispagestyle{empty}%
    \OfficialTOCTitle
    \@starttoc{toc}%
    \clearpage
  \endgroup
  % 正文从第 1 页重新开始
  \pagenumbering{arabic}%
  \setcounter{page}{1}%
}
\makeatother
